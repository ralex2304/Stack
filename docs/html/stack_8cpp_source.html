105int stk_resize(Stack* stk, size_t new_size) {
106 int res = STK_ASSERT(stk);
107
108 if (new_size == 0) {
109 res |= stk->INVALID_CAPACITY;
110 STK_OK(stk, res);
111 return res;
112 }
113
114 Elem_t* tmp = stk->data;
115
116 stk->data = (Elem_t*)recalloc((void*)((Canary_t*)stk->data ON_CANARY_PROTECT(- 1)),
117 // We need to clear right canary and free space before it -> + 1
118 stk->capacity * sizeof(Elem_t) ON_CANARY_PROTECT(+ 1 * sizeof(CANARY_VAL)),
119 calc_stk_data_size(new_size));
120
121 if (!stk->data) {
122 stk->data = tmp;
123 res |= stk->ALLOC_ERR;
124 STK_OK(stk, res);
125 return res;
126 }
127
128 ON_CANARY_PROTECT(stk->data = (Elem_t*)((Canary_t*)stk->data + 1));

173
174 if (stk->size <= 0) {
175 res |= stk->NOTHING_TO_POP;
176 STK_OK(stk, res);
177 return res;
178 }
179
180 *elem = stk->data[--stk->size];
181
182#ifdef HASH_PROTECT
183 stk->data_hash = stk_data_hash_calc(stk);
184 stk->struct_hash = stk_struct_hash_calc(stk);
185#endif //< #ifdef HASH_PROTECT
186
187 ON_DEBUG(stk->data[stk->size] = stk->POISON);
188
189 res |= stk_resize_down(stk);
190
191 res |= STK_VERIFY(stk);
192 STK_OK(stk, res);
193

309int stk_verify(Stack* stk) {
310 assert(stk);
311
312 int res = stk->OK;
313
314 if (!stk_is_initialised(stk)) {
315 res |= stk->UNITIALISED;
316 return res;
317 }
318
319 bool is_data_valid = true;
320
321 if (!is_ptr_valid(stk->data)) {
322 res |= stk->DATA_INVALID_PTR;
323 is_data_valid = false;
324 }
325
326 if (is_data_valid) {
327 for (ssize_t i = 0; i < stk->size; i++)
328 if (stk->data[i] == stk->POISON) {
329 res |= stk->POISON_VAL_FOUND;
330 break;
331 }
332
333 for (ssize_t i = stk->size; i < stk->capacity; i++)
334 if (stk->data[i] != stk->POISON) {
335 res |= stk->NON_POISON_EMPTY;
336 break;
337 }
338 }
339
340#ifdef HASH_PROTECT
341 CRC32_t tmp_struct_hash = stk->struct_hash;
342
343 if (is_data_valid) {
344 CRC32_t data_hash = stk_data_hash_calc(stk);
345 CHECK_ERR_(stk->data_hash != data_hash, stk->DATA_HASH_ERR);
346 }
347
348 CRC32_t struct_hash = stk_struct_hash_calc(stk);
349 stk->struct_hash = tmp_struct_hash;
350
351 CHECK_ERR_(stk->struct_hash != struct_hash, stk->STRUCT_HASH_ERR);
352#endif //< #ifdef HASH_PROTECT
353
354 CHECK_ERR_(stk->capacity < stk->size, stk->LOW_CAPACITY);
355 CHECK_ERR_(stk->capacity < 0, stk->NEGATIVE_CAPACITY);
356 CHECK_ERR_(stk->size < 0, stk->NEGATIVE_SIZE);
357
358#ifdef CANARY_PROTECT
359 CHECK_ERR_(stk->canary_left != CANARY_VAL, stk->STRUCT_L_CANARY_ERR);
360 CHECK_ERR_(stk->canary_right != CANARY_VAL, stk->STRUCT_R_CANARY_ERR);
361
362 if (is_data_valid) {
363 CHECK_ERR_( *left_data_canary_ptr(stk) != CANARY_VAL, stk->DATA_L_CANARY_ERR);
364 CHECK_ERR_(*right_data_canary_ptr(stk) != CANARY_VAL, stk->DATA_R_CANARY_ERR);
365 }
366#endif //< #ifdef CANARY_PROTECT
367
368 return res;
369}

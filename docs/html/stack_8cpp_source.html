292int stk_verify(Stack* stk) {
293 assert(stk);
294
295 int res = stk->OK;
296
297 if (stk->size == stk->UNITIALISED_CAPACITY &&
298 stk->capacity == stk->UNITIALISED_CAPACITY &&
299 stk->data == nullptr) {
300 res |= stk->UNITIALISED;
301 return res;
302 }
303
304 CHECK_ERR_(!is_ptr_valid(stk->data, 'w'), stk->DATA_INVALID_PTR);
305
306 for (ssize_t i = 0; i < stk->size; i++)
307 if (stk->data[i] == stk->POISON) {
308 res |= stk->POISON_VAL_FOUND;
309 break;
310 }
311
312 for (ssize_t i = stk->size; i < stk->capacity; i++)
313 if (stk->data[i] != stk->POISON) {
314 res |= stk->NON_POISON_EMPTY;
315 break;
316 }
317
318#ifdef HASH_PROTECT
319 CRC32_t tmp_struct_hash = stk->struct_hash;
320
321 CRC32_t data_hash = stk_data_hash_calc(stk);
322 CRC32_t struct_hash = stk_struct_hash_calc(stk);
323
324 stk->struct_hash = tmp_struct_hash;
325
326 CHECK_ERR_(stk->struct_hash != struct_hash, stk->STRUCT_HASH_ERR);
327 CHECK_ERR_( stk->data_hash != data_hash, stk->DATA_HASH_ERR);
328#endif // #ifdef HASH_PROTECT
329
330 CHECK_ERR_(stk->capacity < stk->size, stk->LOW_CAPACITY);
331 CHECK_ERR_(stk->capacity < 0, stk->NEGATIVE_CAPACITY);
332 CHECK_ERR_(stk->size < 0, stk->NEGATIVE_SIZE);
333
334#ifdef CANARY_PROTECT
335 CHECK_ERR_(stk->canary_left != CANARY_VAL, stk->STRUCT_L_CANARY_ERR);
336
337 CHECK_ERR_(((Canary_t*)stk->data)[-1] != CANARY_VAL, stk->DATA_L_CANARY_ERR);
338 CHECK_ERR_(*((Canary_t*)(stk->data + stk->capacity)) != CANARY_VAL, stk->DATA_R_CANARY_ERR);
339#endif // #ifdef CANARY_PROTECT
340
341 return res;
342}

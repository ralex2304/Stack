100 ssize_t new_size = stk->capacity;

103 new_size = stk->capacity * 2;

105 new_size = stk->capacity / 2;

301int stk_verify(Stack* stk) {
302 assert(stk);
303
304 int res = stk->OK;
305
306 if (!stk_is_initialised(stk)) {
307 res |= stk->UNITIALISED;
308 return res;
309 }
310
311 bool is_data_valid = true;
312
313 if (!is_ptr_valid(stk->data, 'w')) {
314 res |= stk->DATA_INVALID_PTR;
315 is_data_valid = false;
316 }
317
318 if (is_data_valid)
319 for (ssize_t i = 0; i < stk->size; i++)
320 if (stk->data[i] == stk->POISON) {
321 res |= stk->POISON_VAL_FOUND;
322 break;
323 }
324
325 if (is_data_valid)
326 for (ssize_t i = stk->size; i < stk->capacity; i++)
327 if (stk->data[i] != stk->POISON) {
328 res |= stk->NON_POISON_EMPTY;
329 break;
330 }
331
332#ifdef HASH_PROTECT
333 CRC32_t tmp_struct_hash = stk->struct_hash;
334
335 if (is_data_valid) {
336 CRC32_t data_hash = stk_data_hash_calc(stk);
337 CHECK_ERR_( stk->data_hash != data_hash, stk->DATA_HASH_ERR);
338 }
339
340 CRC32_t struct_hash = stk_struct_hash_calc(stk);
341 stk->struct_hash = tmp_struct_hash;
342
343 CHECK_ERR_(stk->struct_hash != struct_hash, stk->STRUCT_HASH_ERR);
344#endif // #ifdef HASH_PROTECT
345
346 CHECK_ERR_(stk->capacity < stk->size, stk->LOW_CAPACITY);
347 CHECK_ERR_(stk->capacity < 0, stk->NEGATIVE_CAPACITY);
348 CHECK_ERR_(stk->min_capacity == 0, stk->INVALID_MIN_CAPACITY);
349 CHECK_ERR_(stk->size < 0, stk->NEGATIVE_SIZE);
350
351#ifdef CANARY_PROTECT
352 CHECK_ERR_(stk->canary_left != CANARY_VAL, stk->STRUCT_L_CANARY_ERR);
353 CHECK_ERR_(stk->canary_right != CANARY_VAL, stk->STRUCT_R_CANARY_ERR);
354
355 if (is_data_valid) {
356 CHECK_ERR_( *left_data_canary_ptr(stk) != CANARY_VAL, stk->DATA_L_CANARY_ERR);
357 CHECK_ERR_(*right_data_canary_ptr(stk) != CANARY_VAL, stk->DATA_R_CANARY_ERR);
358 }
359#endif // #ifdef CANARY_PROTECT
360
361 return res;
362}

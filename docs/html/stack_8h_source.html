352inline int stk_resize_up(Stack* stk) {
353 int res = STK_ASSERT(stk);
354
355 ssize_t new_cap = stk->capacity;
356
357 while (stk->size >= new_cap)
358 new_cap *= 2;
359
360 if (new_cap != stk->capacity)
361 return res | stk_resize(stk, new_cap);
362
363 return res;
364}

372inline int stk_resize_down(Stack* stk) {
373 int res = STK_ASSERT(stk);
374
375 ssize_t new_cap = stk->capacity;
376
377 if (stk->size * 4 <= new_cap && new_cap >= (ssize_t)stk->DEFAULT_CAPACITY * 2)
378 new_cap /= 2;
379
380 if (new_cap != stk->capacity)
381 return res | stk_resize(stk, new_cap);
382
